//m_select
puts (1..10).m_select { |i|  i % 3 == 0 }   #=> [3, 6, 9]
puts [1,2,3,4,5].m_select { |num|  num.even?  }   #=> [2, 4]
puts [:foo, :bar].m_select { |x| x == :foo }   #=> [:foo]

//m_all?

puts %w[ant bear cat].m_all? { |word| word.length >= 4 }
puts %w[ant bear cat].m_all? { |word| word.length >= 3 }
